/* ========================================================================
   INNER GAME WINDOW (.game-scene-inner)
   ======================================================================== */
/* The inner container has a FIXED 1280x800 size that never changes */
/* This fixed size is critical: all game elements (pot, sink, flame) are
   positioned as percentages relative to this fixed 1280x800 window */
/* 
   Design rationale:
   - Background PNG is exactly 1280x800 pixels
   - Percentage-based positioning (e.g., pot at 50% X, 40% Y) only works correctly
     with a known fixed reference size
   - The outer container resizes with the viewport, but this inner one does not
   - This prevents scaling drift and ensures consistent drag behavior
*/
.game-scene-inner {
  position: relative;           /* Create positioning context for child elements */
  width: 1280px;                /* FIXED: matches background.png dimensions */
  height: 800px;                /* FIXED: matches background.png dimensions */
  background-image: none;
  background-size: 100% 100%;   /* Stretch to fill entire 1280x800 area */
  background-position: center;
  background-repeat: no-repeat;
  border: 2px solid var(--workshop-panel-border, rgba(0, 0, 0, 0.15));
  box-shadow: 0 0 20px rgba(0, 0, 0, 0.5);  /* Add depth with shadow */
}

/* Positioned using percentage-based coordinates (e.g., potX = 50 = 50% across) */
/* IMPORTANT: The pot has a center-based position, not corner-based:
     - potX = 50 means the pot CENTER is at 50% horizontally (640px in 1280px window)
     - potY = 40 means the pot CENTER is at 40% vertically (320px in 800px window)
     - We calculate actual pixel positions as: left = (percentX/100) * 1280, etc.
*/

  position: absolute;
